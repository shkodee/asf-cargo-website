```tsx
tubelight-navbar.tsx
"use client"

import React, { useEffect, useState } from "react"
import { motion } from "framer-motion"
import Link from "next/link"
import { LucideIcon } from "lucide-react"
import { cn } from "@/lib/utils"

interface NavItem {
  name: string
  url: string
  icon: LucideIcon
}

interface NavBarProps {
  items: NavItem[]
  className?: string
}

export function NavBar({ items, className }: NavBarProps) {
  const [activeTab, setActiveTab] = useState(items[0].name)
  const [isMobile, setIsMobile] = useState(false)

  useEffect(() => {
    const handleResize = () => setIsMobile(window.innerWidth < 768)
    handleResize()
    window.addEventListener("resize", handleResize)
    return () => window.removeEventListener("resize", handleResize)
  }, [])

  return (
    <div className={cn("fixed bottom-0 sm:top-0 left-1/2 -translate-x-1/2 z-50 mb-6 sm:pt-6", className)}>
      <div className="flex items-center gap-3 bg-background/5 border border-border backdrop-blur-lg py-1 px-1 rounded-full shadow-lg">
        {items.map((item) => {
          const Icon = item.icon
          const isActive = activeTab === item.name
          return (
            <Link
              key={item.name}
              href={item.url}
              onClick={() => setActiveTab(item.name)}
              className={cn(
                "relative cursor-pointer text-sm font-semibold px-6 py-2 rounded-full transition-colors",
                "text-foreground/80 hover:text-primary",
                isActive && "bg-muted text-primary",
              )}
            >
              <span className="hidden md:inline">{item.name}</span>
              <span className="md:hidden"><Icon size={18} strokeWidth={2.5} /></span>
              {isActive && (
                <motion.div
                  layoutId="lamp"
                  className="absolute inset-0 w-full bg-primary/5 rounded-full -z-10"
                  initial={false}
                  transition={{ type: "spring", stiffness: 300, damping: 30 }}
                >
                  <div className="absolute -top-2 left-1/2 -translate-x-1/2 w-8 h-1 bg-primary rounded-t-full">
                    <div className="absolute w-12 h-6 bg-primary/20 rounded-full blur-md -top-2 -left-2" />
                    <div className="absolute w-8 h-6 bg-primary/20 rounded-full blur-md -top-1" />
                    <div className="absolute w-4 h-4 bg-primary/20 rounded-full blur-sm top-0 left-2" />
                  </div>
                </motion.div>
              )}
            </Link>
          )
        })}
      </div>
    </div>
  )
}

## Equipment scroll animation (non-component prompt)

**Date:** 2026-07-15
**Outcome:** Not a component-integration prompt — a scroll-animation spec that was reworded into a
video-generation prompt for another AI tool. Still **blocked** on that reference video coming back.
See `TODO.md`/`PROJECT_BRIEF.md` for status.

```
A clean, photoreal product-animation sequence for a trucking company website, built from three
reference images: a semi truck (truck.png), a dry van trailer (van.png), and a flatbed trailer
(flatbed.png). Dark navy background (#0c1c34), soft studio lighting, subtle ground shadow under
vehicles to sell weight and contact with the ground. 8–10 second loopable sequence, 16:9.

- 0.0–1.5s: The truck rolls in from off-screen right, wheels visibly rotating, and settles
  centered in frame with a soft bounce/settle on arrival.
- 1.5–2.5s: The van and flatbed trailer fade/scale in at the top-right corner of the frame, small
  scale, van slightly larger and layered in front of the flatbed (flatbed peeking out behind/below
  it). Both have a light rolling-wheel wiggle as they appear.
- 2.5–4.5s: The van grows from its small top-right position and travels down to couple onto the
  back of the truck, forming one continuous truck+trailer silhouette. The flatbed stays parked,
  small, top-right.
- 4.5–6.5s: The van uncouples, shrinks, and exits toward the bottom-right of frame, fading out. At
  the same time, the flatbed grows from its top-right position and couples onto the truck in the
  exact spot the van just left.
- 6.5–8.5s: The flatbed uncouples, shrinks, and exits toward the bottom-right, fading out the same
  way the van did. The truck recenters and holds alone as the final frame — truck only, no trailer.

Motion style: smooth, mechanical, slightly weighted (not bouncy/cartoonish) — think
product-commercial vehicle animation, not toy physics. Consistent lighting and scale logic across
all three vehicles throughout.
```
